local Lib = loadstring(game:HttpGet("https://raw.githubusercontent.com/7yhx/kwargs_Ui_Library/main/source.lua"))()

local UI = Lib:Create{
   Theme = "Dark", -- or any other theme
   Size = UDim2.new(0, 555, 0, 400) -- default
}

local Main = UI:Tab{
   Name = "Main"
}

local Divider = Main:Divider{
   Name = "Main shit"
}

local QuitDivider = Main:Divider{
   Name = "Quit"
}

-- All functions have the Name, Description and Callback arguments so you can use them whenever ig yeah
local KillAll = Divider:Button{
   Name = "PUXAR ARMAS",
   Description = "Kills all the players in the game!",
-- Cria uma função para clonar uma tool
function clonarTool(tool)
  -- Verifica se a tool é válida
  if tool and tool:IsA("Tool") then
    -- Cria uma cópia da tool
    local clone = tool:Clone()
    -- Coloca a cópia no inventário do jogador local
    clone.Parent = game.Players.LocalPlayer.Backpack
    -- Retorna a cópia
    return clone
  else
    -- Retorna nil se a tool não é válida
    return nil
  end
end

-- Cria uma lista vazia para guardar as tools clonadas
local toolsClonadas = {}

-- Percorre todos os players do jogo
for _, player in pairs(game.Players:GetPlayers()) do
  -- Verifica se o player não é o jogador local
  if player ~= game.Players.LocalPlayer then
    -- Obtém a tool que está na mão do player
    local tool = player.Character:FindFirstChildWhichIsA("Tool")
    -- Clona a tool e guarda na lista
    local clone = clonarTool(tool)
    table.insert(toolsClonadas, clone)
  end
end

-- Mostra uma mensagem com o número de tools clonadas
print("Você clonou " .. #toolsClonadas .. " tools dos outros players!")
}
